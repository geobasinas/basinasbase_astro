```python
import math

# Input coefficients
a = float(input("Enter coefficient a: "))
b = float(input("Enter coefficient b: "))
c = float(input("Enter coefficient c: "))

if a == 0:
    # Not a quadratic equation, solve as a linear equation
    if b != 0:
        solution = -c / b
        print(f"Linear equation solution: x = {solution}")
    else:
        print("No solution, since both a and b are 0.")
else:
    # Calculate the discriminant
    discriminant = b**2 - 4*a*c
    if discriminant > 0:
        # Two distinct real roots
        root1 = (-b + math.sqrt(discriminant)) / (2*a)
        root2 = (-b - math.sqrt(discriminant)) / (2*a)
        print(f"Two distinct real solutions: x1 = {root1}, x2 = {root2}")
    elif discriminant == 0:
        # One real root (double root)
        root = -b / (2*a)
        print(f"One real solution: x = {root}")
    else:
        # No real roots, two complex roots
        real_part = -b / (2*a)
        imaginary_part = math.sqrt(-discriminant) / (2*a)
        print(f"Two complex solutions: x1 = {real_part} + {imaginary_part}i, x2 = {real_part} - {imaginary_part}i")
```
